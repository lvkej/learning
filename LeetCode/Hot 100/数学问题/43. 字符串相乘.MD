    public String multiply(String num1, String num2) {
        if("0".equals(num1) || "0".equals(num2)){
            return "0";
        }
        String s1 = new StringBuilder(num1).reverse().toString();
        String s2 = new StringBuilder(num2).reverse().toString();
        int len1 = s1.length();
        int len2 = s2.length();
        int[] arr = new int[len1 + len2];   //len1位*len2位的数结果必在len1+len2或len1+len2-1
        for (int i = 0; i < len1; i++) {
            for (int j = 0; j < len2; j++) {
                int x = s1.charAt(i)-'0';
                int y = s2.charAt(j)-'0';
                arr[i+j]+=x*y;
            }
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < len1 + len2-1; i++) {
            arr[i+1]+=arr[i]/10;
            arr[i]%=10;
            sb.append(arr[i]);
        }
        if(arr[len1+len2-1]!=0){
            sb.append(arr[len1+len2-1]);
        }
        return sb.reverse().toString();
    }
