    public int findPaths(int m, int n, int maxMove, int startRow, int startColumn) {
        final int MOD = 1000000007;
        int count = 0;
        int[][][] dp = new int[maxMove+1][m][n];    //dp[i][j][k]表示i次移动后，移动到[m,n]的路径数
        dp[0][startRow][startColumn] = 1;
        int[][] directions = {{1,0},{-1,0},{0,1},{0,-1}};
        for(int i =1;i<=maxMove;i++){
            for(int row = 0;row<m;row++){
                for(int col = 0;col<n;col++){
                    int curcnt = dp[i-1][row][col];
                    for (int[] direction : directions) {
                        int curRow = row+direction[0];
                        int curCol = col+direction[1];
                        if(curRow>=m||curRow<0||curCol>=n||curCol<0){
                            count = (count+curcnt)%MOD;
                        }else {
                            dp[i][curRow][curCol] = (dp[i][curRow][curCol]+curcnt)%MOD;
                        }
                    }
                }
            }
        }
        return count;

    }
