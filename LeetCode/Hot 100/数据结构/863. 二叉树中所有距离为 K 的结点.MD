    List<Integer> res = new ArrayList();
    List<Integer> contain = new ArrayList();
    public List<Integer> distanceK(TreeNode root, TreeNode target, int k) {
        TreeNode tmp = root;
        Map<Integer,TreeNode> map = new HashMap();
        Deque<TreeNode> list = new LinkedList();
        map.put(root.val,null);
        list.addLast(tmp);
        while(!list.isEmpty()){
            int size = list.size();
            while(size>0){
                TreeNode node = list.pollFirst();
                if(node.left!=null){
                    list.addLast(node.left);
                    map.put(node.left.val,node);
                }
                if(node.right!=null){
                    list.addLast(node.right);
                    map.put(node.right.val,node);
                }
                size--;
            }
        }
        while(target!=null&&k>0){
            dfs(target,k);
            contain.add(target.val);
            target=map.get(target);
            k--;
        }
        return res;
    }

    public void dfs(TreeNode target,int k){
        if(target==null||contain.contains(target.val)){
            return ;
        }
        if(k==0){
            res.add(target.val);
            return;
        }
        dfs(target.left,k--);
        dfs(target.right,k--);
    }
