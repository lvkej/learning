    public TreeNode layerincreate(int ll,int lr,int inl,int inr){
        if(inl>inr){
            return null;
        }

        int i=0,j=0;
        a:for(i = ll;i<=lr;i++){    //从层序遍历的头开始
            for(j =inl;j<=inr;j++){
                if(layer[i]==in[j]){//找到层序的头在中序遍历的下标
                    break a;
                }
            }
        }
        TreeNode node = new TreeNode(layer[i]);
        node.left = layerincreate(ll,lr,inl,j-1);   //保证中序遍历的内容是左节点
        node.right = layerincreate(ll,lr,j+1,inr);  //保证中序遍历的内容是右节点
        return node;
    }
